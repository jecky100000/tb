Params
Numeric up(6450);
Numeric Down(6300);

NumericSeries HHH;
NumericSeries LLL;
Numeric DM_H;

Numeric fff;

HHH = Max(O,C);						//去影线
LLL = Min(O,C);						//去影线

{

	{
		fff = 1;
		Commentary("上挫出");
		flag = 1;
		PlotNumeric("k",HHH[1],HHH,red);
	}
	Else If(LLL<LLL[1])			  			//右包左
	{
		fff =2;
		Commentary("右包左");
		flag = flag+1;
		If(HHH>=HHH[flag])			//向上------------------------------------------------------------------------------------------------2↑
		{
			Commentary("向上");
			ggg = "↑";
			PlotNumeric("k",LLL[1],HHH,red);
		}
		Else						//向下------------------------------------------------------------------------------------------------2↓
		{
			ggg = "↓";
			Commentary("向下");
			PlotNumeric("k",LLL,HHH[1],Green);
		}
	}
	Else //(LLL==LLL[1])		    	//相等---右包左
	{
		fff =3;
		Commentary("右包左");
		flag = flag+1;

		{
			ggg = "↑";

			PlotNumeric("k",LLL,HHH[1],Green);
		}
	}
}
Else If(HHH<HHH[1]) 				//未创新高
{
	If(LLL<LLL[1])							//下挫出
	{
		fff =4;							//-------------------------------------------------------------------------------------------------4
		flag = 1;
		Commentary("下挫出");
		PlotNumeric("k",LLL,LLL[1],Green);
	}
	Else If (LLL>LLL[1])					//左包右
	{
		fff =5;
		Commentary("左包右");
		flag = flag+1;
		If(LLL<=LLL[flag])				//------------------------------------------------------------------------------------------------5↓
		{
			
			ggg = "↓";
			Commentary("向下");
			PlotNumeric("k",LLL[1],LLL[1]+hhh-LLL,Green);
		}
		Else							//------------------------------------------------------------------------------------------------5↑
		{
			ggg = "↑";
			Commentary("向上");
			PlotNumeric("k",LLL,HHH[1],red);
		}
	}
	Else//(LLL==LLL[1])		    		//相等--左包右
	{
		fff =6;							
		Commentary("左包右");
		flag = flag+1;
		If(LLL<LLL[flag])				//------------------------------------------------------------------------------------------------6↓
		{
			
			ggg = "↓";
			Commentary("向下");
			PlotNumeric("k",LLL[1],LLL[1]+hhh-LLL,Green);
		}
		Else							//------------------------------------------------------------------------------------------------6↑
		{
			ggg = "↑";
			Commentary("向上");
			PlotNumeric("k",LLL,LLL+hhh-LLL,red);
		}
	}
}
Else //(HHH==HHH[1])	
{
	If(LLL<LLL[1])							//右包左
	{
		fff =7;
		flag = flag+1;
		If(LLL<LLL[flag])				//------------------------------------------------------------------------------------------------7↓
		{
			ggg = "↓";
			Commentary("向下");
			PlotNumeric("k",LLL,LLL[1],Green);
		}
		Else							//------------------------------------------------------------------------------------------------7↑
		{
			ggg = "↑";
			Commentary("向上");
			PlotNumeric("k",LLL,HHH[1],red);
		}
	}
	Else If (LLL>LLL[1])					//左包右
	{
		fff =8;
		Commentary("左包右");
		flag = flag+1;
		If(LLL<LLL[flag])				//------------------------------------------------------------------------------------------------8↓
		{
			ggg = "↓";
			Commentary("向下");
			PlotNumeric("k",LLL[1],lll[1]+hhh-LLL,Green);
		}
		Else							//------------------------------------------------------------------------------------------------8↑
		{
			ggg = "↑";
			Commentary("向上");
			PlotNumeric("k",LLL,HHH[1],red);
		}
	}
	Else//(LLL==LLL[1])		    		//相等--左包右
	{
		fff =9;
		Commentary("左包右");
		flag = flag+1;
		If(LLL<LLL[flag])//------------------------------------------------------------------------------------------------9↓
		{
			ggg = "↓";
			Commentary("向下");
			PlotNumeric("k",LLL[1],lll[1]+hhh-LLL,Green);
		}
		Else            //------------------------------------------------------------------------------------------------9↑
		{
			ggg = "↑";
			Commentary("向上");
			PlotNumeric("k",LLL,LLL+hhh-LLL,red);
		}
	}
}
PlotString("fff",Text(fff)+ggg,0,IIF(fff == 1,Red,IIF(fff ==4,green,white)));
If (HHH == HHH[1]) PlotNumeric("kh",hhh,0,red);
If (LLL == LLL[1]) PlotNumeric("kl",lll,0,green);
